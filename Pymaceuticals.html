# Put treatments into a list for for loop (and later for plot labels)
drugs = ["Capomulin", "Ramicane", "Infubinol", "Ceftamin"]

# Create empty list to fill with tumor vol data (for plotting)
drug_values = []

for drug in drugs:
    
    # Locate the rows which contain mice on each drug and get the tumor volumes
    drug_df = merged_mouse_study.loc[merged_mouse_study["Drug Regimen"]==drug]
    
    # add subset
    final_volume = drug_df.loc[drug_df["Timepoint"] == drug_df["Max TimePoint"]]
    
    volume_values = final_volume["Tumor Volume (mm3)"]
    
#Append the empty list with Volume Values
    drug_values.append(volume_values)

# Calculate the IQR and quantitatively determine if there are any potential outliers. 
    quartiles = volume_values.quantile([.25,.5,.75])
    lowerq = quartiles[.25]
    upperq = quartiles[.75]
    iqr = upperq - lowerq
   
    # Determine possible outliers using upper and lower bounds
    lower_bound = lowerq -(1.5*iqr)
    upper_bound = upperq +(1.5*iqr)
    
    #Find any outliers in the data
    possible_outliers = (volume_values.loc[(final_volume["Tumor Volume (mm3)"] >= upper_bound) | (final_volume["Tumor Volume (mm3)"] <= lower_bound)]).count()
    
    print(f"{drug} IQR is {iqr}")
    print(f"{drug} Lower Bound is {lower_bound}")
    print(f"{drug} Upper Bound is {upper_bound}")
    print(f"{drug} Outliers are {possible_outliers}")

# Calculate the correlation coefficient and linear regression model 
# for mouse weight and average tumor volume for the Capomulin regimen
x_values = final_df["Weight (g)"]
y_values = final_df["Average Tumor Volume"]

#Find Line regression values to create line equation
(slope, intercept, rvalue, pvalue, stderr) = linregress(x_values,y_values)
regress_values = x_values*slope + intercept
line_eq = "y =" + str(round(slope,2)) + "x +" + str(round(intercept, 2))

#Find the Correlation Coefficient
correlation = st.pearsonr(x_values,y_values)
print(f"The Correlation between Weight(g) and Average Tumor Volume(mm3) is {round(correlation[0],2)}")

#Create Scatter plot again with regression line
plt.scatter(x_values,y_values)
plt.plot(x_values, regress_values, "r-")
plt.annotate(line_eq, (21,36), fontsize="14", color="red")

#Set title and labels
plt.title("Avg. Tumor Vol. to Weight (With Regression line)")
plt.xlabel("Weight(g)")
plt.ylabel("Average Tumor Volume(mm3)")
